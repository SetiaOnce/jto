INSTALL PYTHON
=======================
1. sudo apt install -y build-essential zlib1g-dev libncurses5-dev libgdbm-dev
2. cd /usr/src
3. sudo curl -O https://www.python.org/ftp/python/3.10.11/Python-3.10.11.tgz
4. sudo tar -xzf Python-3.10.11.tgz
5. cd Python-3.10.11
6. sudo ./configure --enable-optimizations
7. sudo make altinstall
8. python3.10 --version
9. sudo update-alternatives --install /usr/bin/python python /usr/local/bin/python3.10 1
10. python --version

INSTALL MYSQL CLIENT
=======================
1. sudo apt update
2. sudo apt install -y python3-dev default-libmysqlclient-dev build-essential pkg-config
3. python3 -m pip install mysqlclient
4. pip install mysqlclient

MEMBUAT SERVICE UBUNTU
=======================
1. sudo nano /etc/systemd/system/app_penimbangan.service
2. Masukkan isi file dibawah ini 
- untuk mendapatkan User & Group jalankan whoami di terminal
- untuk mendapatkan path jalankan pwd di terminal
- untuk mendapatkan lokasi python jalankan which python di terminal

[Unit]
Description=APP PENIMBANGAN
After=network.target

[Service]
User=server
Group=server
WorkingDirectory=/home/server/jto/penimbangan
ExecStart=/home/server/jto/penimbangan/venv/bin/python manage.py runserver 0.0.0.0:8021
Restart=always
RestartSec=5    

[Install]
WantedBy=multi-user.target 

3. sudo systemctl daemon-reload
4. sudo systemctl enable app_penimbangan.service
5. sudo systemctl start app_penimbangan.service
6. sudo systemctl status app_penimbangan.service
